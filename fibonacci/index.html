import java.util.Scanner; public class Fibonacci { public static int fibonacciBottomUp(int n) { if (n <= 1) { return n; } int[] fibArray = new int[n + 1]; fibArray[0] = 0; fibArray[1] = 1; for (int i = 2; i <= n; i++) { fibArray[i] = fibArray[i - 1] + fibArray[i - 2]; } return fibArray[n]; } public static void main(String[] args) { Scanner scanner = new Scanner(System.in); int n = scanner.nextInt(); // Compute Fibonacci value int result = fibonacciBottomUp(n); System.out.println(result); // Close the scanner scanner.close(); } }
